#6.1 Write a script that will count how many games "Cedric Vega" has played in during his entire football career.

>> CREATE VIEW vega_games AS SELECT count(*) AS tGames FROM football_players INNER JOIN football_games ON name='Cedric Vega' AND id=any(players);
   SELECT * FROM vega_games;

>> CREATE VIEW vega_rush AS SELECT (CAST(rushing_yards AS decimal)/CAST(tGames AS decimal)) FROM vega_games, football_players WHERE name = 'Cedric Vega';
